1. 리눅스용 Docker Desktop 설치

  • Docker의 패키지 저장소 를 설정
  • 최신 DEB 패키지 를 다운로드
  • apt로 패키지를 설치

< Docker 패키지 저장소 설정 >



        # 설치에 필요한 패키지 설치

        sudo apt-get update
        sudo apt-get install \
          ca-certificates \
          curl \
          gnupg \
          lsb-release 

        # Docker의 공식 GPG 키 추가

        sudo mkdir -p /etc/apt/keyrings
        curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /etc/apt/keyrings/docker.gpg 

        # 리포지토리를 설정

        # 원문의 $(lsb_release -cs) 부분을 jammy 로 바꿔줘야 한다
        # 리눅스민트의 VERSION_CODENAME=vanessa 를 찾는데
        # 우분투의 UBUNTU_CODENAME=jammy 를 강제로 지정해 줘야
        # 제대로된 저장소를 찾는다 <- 이거 찾느라 힘들었다 

        echo \
        "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/ubuntu \
        jammy stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

< 최신 DEB 패키지 를 다운로드 >

# 참고 자료에 링크 첨부

< 패키지를 설치 >



          sudo apt-get update
          sudo apt-get install ./docker-desktop-<version>-<arch>.deb
